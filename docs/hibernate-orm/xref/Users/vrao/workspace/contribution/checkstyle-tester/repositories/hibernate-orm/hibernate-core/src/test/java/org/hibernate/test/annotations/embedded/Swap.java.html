1   package org.hibernate.test.annotations.embedded;
2   import javax.persistence.AttributeOverride;
3   import javax.persistence.Column;
4   import javax.persistence.Embeddable;
5   import javax.persistence.Embedded;
6   
7   /**
8    * Interest Rate Swap with Tenor. Used here to compose
9    * a swap spread deal.
10   */
11  @Embeddable
12  public class Swap {
13  
14  	/**
15  	 * Tenor (duration) of the swap (in years).
16  	 */
17  	private int tenor;
18  
19  	public int getTenor() {
20  		return tenor;
21  	}
22  
23  	public void setTenor(int tenor) {
24  		this.tenor = tenor;
25  	}
26  
27  	/**
28  	 * Fixed leg (cash flows with the fixed rate).
29  	 */
30  	private FixedLeg fixedLeg;
31  
32  	/**
33  	 * Floating leg (cash flows bound to a financial index).
34  	 */
35  	private FloatLeg floatLeg;
36  
37  	@Embedded
38  	// We retain this annotation to test the precedence of @AttributeOverride
39  	// Outermost override annotation should win
40  	@AttributeOverride(name = "paymentFrequency", column = @Column(name = "FIXED_FREQENCY"))
41  	public FixedLeg getFixedLeg() {
42  		return fixedLeg;
43  	}
44  
45  	public void setFixedLeg(FixedLeg fixedLeg) {
46  		this.fixedLeg = fixedLeg;
47  	}
48  
49  	@Embedded
50  	public FloatLeg getFloatLeg() {
51  		return floatLeg;
52  	}
53  
54  	public void setFloatLeg(FloatLeg floatLeg) {
55  		this.floatLeg = floatLeg;
56  	}
57  }
