1   package org.hibernate.envers;
2   import java.lang.annotation.Retention;
3   import java.lang.annotation.Target;
4   
5   import static java.lang.annotation.ElementType.FIELD;
6   import static java.lang.annotation.ElementType.METHOD;
7   import static java.lang.annotation.ElementType.TYPE;
8   import static java.lang.annotation.RetentionPolicy.RUNTIME;
9   
10  /**
11   * The {@code AuditingOverrides} annotation is used to override the auditing
12   * behavior for one ore more fields (or properties) inside an embedded
13   * component.
14   *
15   * @author Erik-Berndt Scheper
16   * @see javax.persistence.Embedded
17   * @see javax.persistence.Embeddable
18   * @see javax.persistence.MappedSuperclass
19   * @see javax.persistence.AssociationOverride
20   * @see javax.persistence.AssociationOverrides
21   * @see AuditJoinTable
22   * @see AuditOverride
23   */
24  @Target({ TYPE, METHOD, FIELD })
25  @Retention(RUNTIME)
26  public @interface AuditOverrides {
27  	/**
28  	 * @return An array of {@link AuditOverride} values, to define the new auditing
29  	 * behavior.
30  	 */
31  	AuditOverride[] value();
32  }
