1   package org.hibernate.envers.configuration.metadata;
2   import java.util.HashSet;
3   import java.util.Set;
4   
5   import org.hibernate.MappingException;
6   
7   /**
8    * A register of all audit entity names used so far.
9    * @author Adam Warski (adam at warski dot org)
10   */
11  public class AuditEntityNameRegister {
12      private final Set<String> auditEntityNames = new HashSet<String>();
13  
14      /**
15       * @param auditEntityName Name of the audit entity.
16       * @return True if the given audit entity name is already used.
17       */
18      private boolean check(String auditEntityName) {
19          return auditEntityNames.contains(auditEntityName);
20      }
21  
22      /**
23       * Register an audit entity name. If the name is already registered, an exception is thrown.
24       * @param auditEntityName Name of the audit entity.
25       */
26      public void register(String auditEntityName) {
27          if (auditEntityNames.contains(auditEntityName)) {
28              throw new MappingException("The audit entity name '" + auditEntityName + "' is already registered.");
29          }
30          
31          auditEntityNames.add(auditEntityName);
32      }
33  
34      /**
35       * Creates a unique (not yet registered) audit entity name by appending consecutive numbers to the base
36       * name. If the base name is not yet used, it is returned unmodified.
37  	 *
38       * @param baseAuditEntityName The base entity name.
39  	 *
40       * @return A unique audit entity name
41       */
42      public String createUnique(final String baseAuditEntityName) {
43          String auditEntityName = baseAuditEntityName;
44          int count = 1;
45          while (check(auditEntityName)) {
46              auditEntityName = baseAuditEntityName + count++;
47          }
48  
49          return auditEntityName;
50      }
51  }
