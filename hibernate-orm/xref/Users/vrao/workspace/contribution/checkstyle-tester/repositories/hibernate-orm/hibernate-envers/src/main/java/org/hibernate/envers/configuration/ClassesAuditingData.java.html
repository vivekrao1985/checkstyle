1   package org.hibernate.envers.configuration;
2   import java.util.Collection;
3   import java.util.HashMap;
4   import java.util.LinkedHashMap;
5   import java.util.Map;
6   
7   import org.jboss.logging.Logger;
8   
9   import org.hibernate.MappingException;
10  import org.hibernate.envers.configuration.metadata.reader.ClassAuditingData;
11  import org.hibernate.envers.configuration.metadata.reader.PropertyAuditingData;
12  import org.hibernate.envers.internal.EnversMessageLogger;
13  import org.hibernate.envers.tools.MappingTools;
14  import org.hibernate.mapping.PersistentClass;
15  
16  /**
17   * A helper class holding auditing meta-data for all persistent classes.
18   * @author Adam Warski (adam at warski dot org)
19   */
20  public class ClassesAuditingData {
21  
22      public static final EnversMessageLogger LOG = Logger.getMessageLogger(EnversMessageLogger.class, ClassesAuditingData.class.getName());
23  
24      private final Map<String, ClassAuditingData> entityNameToAuditingData = new HashMap<String, ClassAuditingData>();
25      private final Map<PersistentClass, ClassAuditingData> persistentClassToAuditingData = new LinkedHashMap<PersistentClass, ClassAuditingData>();
26  
27      /**
28       * Stores information about auditing meta-data for the given class.
29       * @param pc Persistent class.
30       * @param cad Auditing meta-data for the given class.
31       */
32      public void addClassAuditingData(PersistentClass pc, ClassAuditingData cad) {
33          entityNameToAuditingData.put(pc.getEntityName(), cad);
34          persistentClassToAuditingData.put(pc, cad);
35      }
36  
37      /**
38       * @return A collection of all auditing meta-data for persistent classes.
39       */
40      public Collection<Map.Entry<PersistentClass, ClassAuditingData>> getAllClassAuditedData() {
41          return persistentClassToAuditingData.entrySet();
42      }
43  
44      /**
45       * @param entityName Name of the entity.
46       * @return Auditing meta-data for the given entity.
47       */
48      public ClassAuditingData getClassAuditingData(String entityName) {
49          return entityNameToAuditingData.get(entityName);
50      }
51  
52      /**
53       * After all meta-data is read, updates calculated fields. This includes:
54       * <ul>
55       * <li>setting {@code forceInsertable} to {@code true} for properties specified by {@code @AuditMappedBy}</li>
56       * </ul>
57       */
58      public void updateCalculatedFields() {
59          for (Map.Entry<PersistentClass, ClassAuditingData> classAuditingDataEntry : persistentClassToAuditingData.entrySet()) {
60              PersistentClass pc = classAuditingDataEntry.getKey();
61              ClassAuditingData classAuditingData = classAuditingDataEntry.getValue();
62              for (String propertyName : classAuditingData.getPropertyNames()) {
63                  PropertyAuditingData propertyAuditingData = classAuditingData.getPropertyAuditingData(propertyName);
64                  // If a property had the @AuditMappedBy annotation, setting the referenced fields to be always insertable.
65                  if (propertyAuditingData.getAuditMappedBy() != null) {
66                      String referencedEntityName = MappingTools.getReferencedEntityName(pc.getProperty(propertyName).getValue());
67  
68                      ClassAuditingData referencedClassAuditingData = entityNameToAuditingData.get(referencedEntityName);
69  
70                      forcePropertyInsertable(referencedClassAuditingData, propertyAuditingData.getAuditMappedBy(),
71                              pc.getEntityName(), referencedEntityName);
72  
73                      forcePropertyInsertable(referencedClassAuditingData, propertyAuditingData.getPositionMappedBy(),
74                              pc.getEntityName(), referencedEntityName);
75                  }
76              }
77          }
78      }
79  
80      private void forcePropertyInsertable(ClassAuditingData classAuditingData, String propertyName,
81                                           String entityName, String referencedEntityName) {
82          if (propertyName != null) {
83              if (classAuditingData.getPropertyAuditingData(propertyName) == null) {
84                  throw new MappingException("@AuditMappedBy points to a property that doesn't exist: " +
85                      referencedEntityName + "." + propertyName);
86              }
87  
88              LOG.debugf("Non-insertable property %s.%s will be made insertable because a matching @AuditMappedBy was found in the %s entity",
89                         referencedEntityName,
90                         propertyName,
91                         entityName);
92  
93              classAuditingData
94                      .getPropertyAuditingData(propertyName)
95                      .setForceInsertable(true);
96          }
97      }
98  }
